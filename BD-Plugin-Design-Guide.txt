STANDARD JAVASCRIPT FORBEDRINGER
-------------------------------

```javascript
jQuery(document).ready(function($) {
    // Smooth form submission feedback
    $("form").submit(function() {
        var $button = $(this).find(".button-primary");
        $button.addClass("bd-loading").prop("disabled", true);
        
        var originalText = $button.text();
        $button.text("üíæ Lagrer...");
        
        // Reset after 3 seconds if no page reload
        setTimeout(function() {
            $button.removeClass("bd-loading").prop("disabled", false);
            $button.text(originalText);
        }, 3000);
    });
    
    // Auto-hide success messages
    setTimeout(function() {
        $(".bd-success-notice").fadeOut(500);
    }, 5000);
    
    // Enhanced hover effects
    $(".bd-settings-card").hover(
        function() { 
            $(this).css("transform", "translateY(-2px)"); 
        },
        function() { 
            $(this).css("transform", "translateY(0)"); 
        }
    );
    
    // Toggle sections
    $('[data-toggle]').click(function() {
        var target = $(this).data('toggle');
        $(target).slideToggle(300);
    });
    
    // Smooth scrolling for anchor links
    $("a[href^='#']").click(function(e) {
        var target = $(this.hash);
        if (target.length) {
            e.preventDefault();
            $("html, body").animate({
                scrollTop: target.offset().top - 32
            }, 500);
        }
    });
    
    // Form validation visual feedback
    $("input[required], textarea[required]").blur(function() {
        if ($(this).val() === '') {
            $(this).css('border-color', '#dc3232');
        } else {
            $(this).css('border-color', '#46b450');
        }
    });
    
    // Tooltip functionality
    $('[data-tooltip]').hover(function() {
        var tooltip = $(this).data('tooltip');
        $('<div class="bd-tooltip">' + tooltip + '</div>').appendTo('body').fadeIn();
    }, function() {
        $('.bd-tooltip').remove();
    });
});
```

MODERNE ANIMASJONER OG OVERGANGER
================================================================================

STANDARD TRANSITIONS
--------------------
```css
/* Global smooth transitions */
*, *::before, *::after {
    transition: all 0.3s ease;
}

/* Hover effects */
.bd-hover-lift:hover {
    transform: translateY(-2px);
    box-shadow: 0 6px 25px rgba(0,0,0,0.15);
}

.bd-hover-scale:hover {
    transform: scale(1.02);
}
```

KEYFRAME ANIMASJONER
-------------------
```css
@keyframes fadeInUp {
    from {
        opacity: 0;
        transform: translateY(20px);
    }
    to {
        opacity: 1;
        transform: translateY(0);
    }
}

@keyframes slideIn {
    from {
        opacity: 0;
        transform: translateX(-20px);
    }
    to {
        opacity: 1;
        transform: translateX(0);
    }
}

@keyframes bd-spin {
    0% { transform: rotate(0deg); }
    100% { transform: rotate(360deg); }
}

/* Loading states */
.bd-loading {
    opacity: 0.7 !important;
    pointer-events: none !important;
    position: relative !important;
}

.bd-loading::after {
    content: '' !important;
```

================================================================================
RESPONSIV DESIGN
================================================================================

BREAKPOINTS
----------
- Mobile: max-width: 480px
- Tablet: max-width: 768px
- Desktop: max-width: 1200px
- Large Desktop: min-width: 1201px

RESPONSIVE PATTERNS
------------------
```css
/* Mobile First Approach */
@media screen and (max-width: 782px) {
    .bd-admin-header {
        flex-direction: column;
        text-align: center;
        gap: 20px;
        padding: 30px 20px;
    }
    
    .nav-tab {
        padding: 12px 15px;
        font-size: 14px;
    }
    
    .tab-content {
        padding: 20px;
    }
    
    .form-table th {
        width: auto;
        display: block;
        padding: 10px 0 5px 0;
    }
    
    .form-table td {
        display: block;
        padding: 5px 0 20px 0;
    }
    
    .form-table .regular-text,
    .form-table .large-text {
        width: 100%;
    }
}

@media (max-width: 480px) {
    .bd-admin-header {
        padding: 25px 15px;
    }
    
    .bd-branding h2 {
        font-size: 1.8em;
    }
}
```
